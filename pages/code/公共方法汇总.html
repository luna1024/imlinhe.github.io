// 定义函数，在其原型上添加方法
function commonJs() {

}

commonJs.prototype = {
  /**
   * 加载单个外部js文件
   */
  loadSingleScript: function(src, callback) {
    var s = document.createElement('script');
    s.async = false;
    s.src = src;
    s.addEventListener('load', function () {
        callback();
    }, false);
    document.body.appendChild(s);
  },
  /**
   * rem转成px
   */
  remToPx: function (remNum) {
    let rootFontSize = Number.parseFloat(document.documentElement.style.fontSize);
    return remNum * rootFontSize;
  },
  /**
   * 获取url参数
   * @param param 参数名
   * @returns {*}
   */
  getUrlParams: function (param) {
    var ps = decodeURI(location.href)
    if (ps == '') return ''
    var params = (ps.substr(ps.lastIndexOf("?") + 1)).split("&")
    if (params != null) {
      for (var i = 0; i < params.length; i++) {
        var strs = params[i].split("=")
        if (strs[0] == param && strs[1]) {
          return strs[1]
        }
      }
    }
    return ""
  },
  /**
   * 删除url参数
   * @param name 参数名
   * @returns url
   */
  delUrlParams: function(name){
    var href = location.href;
    var base = href.split('?')[0];
    var query = href.split('?')[1];
    if (!query) {
      return href
    }
    if (query.indexOf(name)>-1) {
        var obj = {}
        var arr = query.split("&");
        for (var i = 0; i < arr.length; i++) {
            arr[i] = arr[i].split("=");
            obj[arr[i][0]] = arr[i][1];
        };
        delete obj[name];
        if (JSON.stringify(obj) != '{}') {
          base += '?'
        }
        var url = base + JSON.stringify(obj).replace(/[\"\{\}]/g,"").replace(/\:/g,"=").replace(/\,/g,"&");
        return url
    } else {
      return href
    }
  },
   /**
   * 替换url参数
   * @param name 参数名
   * @returns url
   */
  repUrlParams: function(key, value){
    var href = location.href;
    var base = href.split('?')[0];
    var query = href.split('?')[1];
    if (!query) {
      return href
    }
    if (query.indexOf(key)>-1) {
        var obj = {}
        var arr = query.split("&");
        for (var i = 0; i < arr.length; i++) {
            arr[i] = arr[i].split("=");
            obj[arr[i][0]] = arr[i][1];
        };
        obj[key] = value;
        if (JSON.stringify(obj) != '{}') {
          base += '?'
        }
        var url = base + JSON.stringify(obj).replace(/[\"\{\}]/g,"").replace(/\:/g,"=").replace(/\,/g,"&");
        return url
    } else {
      return href
    }
  },
  /**
   * 设备类型
   */
  getPhoneType() {
    var result = false;
    var u = navigator.userAgent;
    if (u.indexOf('Android') > -1 || u.indexOf('Linux') > -1) {//安卓手机
      result = false;
    } else if (u.indexOf('iPhone') > -1) {//苹果手机
      result = true;
    } else if (u.indexOf('Windows Phone') > -1) {//winphone手机
      result = false;
    } else if (u.indexOf('iPhone Simulator') > -1) {
      result = true;
    }
    return result;
  },
  /**
   * 转换时间格式的过滤器
   * 用法：
   * times = 1489895605000
   * {{times | filterFormatDate('yyyy-MM-dd hh:mm:ss')}}
   * @param time 毫秒数
   * @param fmt 要转换的时间格式
   */
  filterFormatDate: function (time, fmt) {
    var d = new Date(Number(time));
    var o = {
      "M+": d.getMonth() + 1,                 //月份
      "d+": d.getDate(),                    //日
      "h+": d.getHours(),                   //小时
      "m+": d.getMinutes(),                 //分
      "s+": d.getSeconds(),                 //秒
      "q+": Math.floor((d.getMonth() + 3) / 3), //季度
      "S": d.getMilliseconds()             //毫秒
    };
    if (/(y+)/.test(fmt))
      fmt = fmt.replace(RegExp.$1, (d.getFullYear() + "").substr(4 - RegExp.$1.length));
    for (var k in o)
      if (new RegExp("(" + k + ")").test(fmt))
        fmt = fmt.replace(RegExp.$1, (RegExp.$1.length == 1) ? (o[k]) : (("00" + o[k]).substr(("" + o[k]).length)));
    return fmt;
  },
  /**
   * 格式化倒计时
   * @param time 时间戳
   * return [[0,0], [0,0], [0,0], [1]]
   */
  formatCountDown: function (time) {
    var h = 0, m = 0, s = 0, ms = 0;
    if (time >= 0) {
      h = Math.floor(time / 1000 / 60 / 60);
      m = Math.floor(time / 1000 / 60 % 60);
      s = Math.floor(time / 1000 % 60);
      ms = Math.floor(time % 1000);
    }
    h = this.parseDate(h).toString().split('');
    m = this.parseDate(m).toString().split('');
    s = this.parseDate(s).toString().split('');
    ms = [Math.floor(ms/100)]
    return {
      h: h,
      m: m,
      s: s,
      ms: ms
    }
  },
  /**
   * 将小于10的数字前加一个0
   * @param t 需要转换的数字
   * @returns 返回转换过的数字
   */
  parseDate: function (t) {
    return t < 10 ? "0" + t : t;
  },
  /**
   * 是否是微信浏览器，true代表是微信浏览器
   */
  isWeiXin() {
    const ua = window.navigator.userAgent.toLowerCase();
    return ua.match(/MicroMessenger/i) == 'micromessenger'
  },
  /**
   * 获取cookie的函数
   */
  getCookie(cookieName) {
    var strCookie = document.cookie;
    var arrCookie = strCookie.split("; ");
    for (var i = 0; i < arrCookie.length; i++) {
      var arr = arrCookie[i].split("=");
      if (cookieName == arr[0]) {
        return arr[1];
      }
    }
    return "";
  },
  //设置 setCookie
  setCookie(key,val,time){
    var date=new Date(); //获取当前时间
    var expiresDays=time;  //将date设置为n天以后的时间
    date.setTime(date.getTime()+expiresDays*24*3600*1000); //格式化为cookie识别的时间
    document.cookie=key + "=" + val +";expires="+date.toGMTString();  //设置cookie
  },
    //节流函数
  throttle(fn, delay) {
    //now:当前的时间; lastExec:上次执行的时间; timer: 记录timeout的id; context: fn执行的上下文作用域；args:函数执行传递的参数
    var now, lastExec, timer, context, args;
    //scroll回调函数真正执行的核心函数
    var execute = function() {
        fn.apply(context, args);
        lastExec = now;
    };
    //闭包函数(绑定到scroll事件上的回调函数)
    return function() {
      context = this;
      args = arguments;
      now = Date.now();
      if (timer) {
        clearTimeout(timer);
        timer = null;
      }
      if (lastExec) {
        //判断是否超过指定时间间隔，超过则执行
        var diff = delay - (now - lastExec);
        if (diff < 0) {
          execute();
        } else {
          timer = setTimeout(() => {
              execute();
          }, diff);
        }
      } else {
        execute();
      }
    }
  },
  /**
   * 防抖函数
   * @param {*} func 传入的需要防抖的函数
   * @param {*} wait 等待时间
   */
  debounce(func, wait = 500) {
    let timer = 0
    return function(...args) {
      clearTimeout(timer)
      timer = setTimeout(() => {
        func.apply(this, args)
      }, wait)
    }
  },
  /**
   * 深拷贝...(避免引用类型)
   * @params obj {object} 需要拷贝的对象
   * @return {object} 返回一个新的对象
   * */
  copyObject: function(obj) {
    if (!obj) {
      return ''
    }
    var tempBox
    if (Array.isArray(obj)) {
      tempBox = []
    } else {
      tempBox = {}
    }
    for (var key in obj) {
      if (typeof(obj[key]) != 'object') {
        tempBox[key] = obj[key]
      } else {
        tempBox[key] = this.copyObject(obj[key])
      }
    }
    return tempBox
  },
  /**
   * 复制文本
   **/
  copyText(content){
    let aux = window.document.createElement("input")
    aux.setAttribute("value", content)
    window.document.body.appendChild(aux)
    // 复制内容
    aux.select()
    // 将内容复制到剪贴板
    window.document.execCommand("copy")
    window.document.body.removeChild(aux)
    //提示...
  },
}
